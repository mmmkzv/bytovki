  <!-- JavaScript for Filtering Catalog -->

  <script>

    function filterCatalog() {

      const type = document.getElementById('type').value;

      const size = document.getElementById('size').value;

      const price = document.getElementById('price').value;

      const catalogItems = document.querySelectorAll('#catalog-items .bg-gray-100');


      catalogItems.forEach(item => {

        const itemType = item.getAttribute('data-type');

        const itemSize = item.getAttribute('data-size');

        const itemPrice = item.getAttribute('data-price');


        if (

          (type === 'Тип бытовки' || type === itemType) &&

          (size === 'Размер' || size === itemSize) &&

          (price === 'Цена' || (parseInt(itemPrice) <= parseInt(price.replace(' ₽', '').replace('свыше', ''))))

        ) {

          item.style.display = 'block';

        } else {

          item.style.display = 'none';

        }

      });

    }


    // Constructor form submission handler

    document.getElementById('constructor-form').addEventListener('submit', function(e) {

      e.preventDefault();

      const type = document.getElementById('constructor-type').value;

      const size = document.getElementById('constructor-size').value;

      const color = document.getElementById('constructor-color').value;

      const options = document.getElementById('constructor-options').value;


      document.getElementById('constructor-result').innerHTML = `

        <p><strong>Тип бытовки:</strong> ${type}</p>

        <p><strong>Размер:</strong> ${size}</p>

        <p><strong>Цвет отделки:</strong> ${color}</p>

        <p><strong>Дополнительные опции:</strong> ${options}</p>

        <p><strong>Рассчитанная стоимость:</strong> 200 000 ₽ (примерная)</p>

      `;

    });

  </script>
